🌈 Flexbox 布局详解

Flex 容器属性

.container {
  display: flex;              /* 启用flex布局 */
  flex-direction: row;        /* 主轴方向：row | column */
  justify-content: center;    /* 主轴对齐：flex-start | center | space-between */
  align-items: center;        /* 交叉轴对齐：flex-start | center | stretch */
  flex-wrap: wrap;            /* 换行：nowrap | wrap */
}

Flex 项目属性

.item {
  flex: 1;                    /* flex-grow flex-shrink flex-basis */
  order: 1;                   /* 排列顺序 */
  align-self: center;         /* 单独对齐 */
}

margin: auto 的魔法

.footer {
  margin-top: auto;           /* 在flex column容器中推到底部 */
}

.centered-item {
  margin: auto;               /* 在flex容器中完全居中 */
}

🚀 JavaScript 基础语法

变量声明

// 现代变量声明
let 变量 = "可变的值";
const 常量 = "不可变的值";

// 数据类型
let 数字 = 42;
let 字符串 = "Hello World";
let 布尔值 = true;
let 数组 = [1, 2, 3, "香草"];
let 对象 = {
  名字: "香草",
  年龄: 18,
  技能: ["卖萌", "写代码"]
};

条件语句

let 饥饿值 = 80;

if (饥饿值 > 70) {
  console.log("香草饿了，要吃小鱼干喵！");
} else if (饥饿值 > 30) {
  console.log("香草有点饿了喵～");
} else {
  console.log("香草饱饱的，要睡觉觉喵～");
}

循环语句

// for循环
for (let i = 0; i < 3; i++) {
  console.log("第" + (i+1) + "次撒娇");
}

// 数组遍历
let 技能 = ["卖萌", "写代码", "睡觉"];
技能.forEach(function(技能名) {
  console.log("香草会" + 技能名);
});

函数

// 函数声明
function 撒娇() {
  console.log("喵～主人抱抱～");
}

// 带参数的函数
function 吃零食(食物) {
  return "香草开心地吃" + 食物;
}

// 箭头函数 (ES6)
const 计算年龄 = (人类年龄) => 人类年龄 * 7;

🎯 DOM 操作

选择元素

// 通过ID选择
let 元素 = document.getElementById("myId");

// 通过选择器选择
let 标题 = document.querySelector("h1");
let 按钮们 = document.querySelectorAll(".button");

// 通过类名选择
let 类元素 = document.getElementsByClassName("myClass");

修改元素

// 修改内容
元素.textContent = "新文本";
元素.innerHTML = "<strong>HTML内容</strong>";

// 修改样式
元素.style.color = "purple";
元素.style.fontSize = "20px";

// 修改属性
元素.setAttribute("class", "新的类名");
元素.classList.add("新类名");
元素.classList.remove("旧类名");

事件处理

// 添加事件监听器
按钮.addEventListener("click", function() {
  alert("喵～主人点我啦！");
});

// 简写形式
按钮.onclick = function() {
  console.log("被点击了");
};

// 箭头函数形式
按钮.addEventListener("click", () => {
  console.log("箭头函数事件");
});

🌐 网页发布指南

发布方式对比

GitHub Pages 🌟

优点：免费、稳定、与代码管理集成

缺点：需要学习Git基础

适用：个人项目、学习作品

Netlify 🚀

优点：简单易用、拖拽上传、自动部署

缺点：免费版有流量限制

适用：快速部署、静态网站

Vercel ⚡

优点：现代化、支持多种框架、全球CDN

缺点：功能复杂、学习成本较高

适用：专业项目、团队协作

发布步骤 (以GitHub Pages为例)

1. 注册GitHub账号
2. 创建新的代码仓库
3. 上传网页文件到仓库
4. 在仓库设置中启用GitHub Pages
5. 获得专属网址分享给朋友
